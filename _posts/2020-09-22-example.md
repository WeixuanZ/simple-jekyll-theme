```python
def sieve(n): 
    primes = [False] * 2 + [True] * (n-2)

    i = 2
    while i * i < n:
        if primes[i] is True:
            for j in range(i * i, n, i):
                primes[j] = False
        i += 1
    
    return [indx for indx, state in enumerate(primes) if state is True]
```
